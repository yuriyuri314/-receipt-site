  const defaultItems = [

  }

  const storeNameEl = document.getElementById('storeName');

  const autoSubline = document.getElementById('autoSubline');

  const autoFooterTitle = document.getElementById('autoFooterTitle');

  const autoReceiptCode = document.getElementById('autoReceiptCode');

  const autoFooterCode = document.getElementById('autoFooterCode');

  const barcodeText = document.getElementById('barcodeText');

  const barcodeEl = document.getElementById('barcode');


  function sanitizeStoreName(value) {

    const clean = value.replace(/\n+/g, ' ').trim();

    return clean || 'UNKNOWN';

  }


  function slugify(value) {

    return sanitizeStoreName(value)

      .toUpperCase()

      .replace(/[^A-Z0-9가-힣 ]/g, '')

      .replace(/\s+/g, '-');

  }


  function makeBarcodeNumber(source) {

    const text = sanitizeStoreName(source).toUpperCase();

    let hash = 0;

    for (let i = 0; i < text.length; i++) {

      hash = ((hash << 5) - hash + text.charCodeAt(i)) >>> 0;

    }

    const digits = String(hash).padStart(10, '0').slice(0, 10);

    return '88' + digits;

  }


  function buildBarcodeBars(digits) {

    barcodeEl.innerHTML = '';

    const start = [2,1,2,2,2,2];

    const end = [2,3,3,1,1,1,2];

    const map = {

      '0':[3,2,1,1], '1':[2,2,2,1], '2':[2,1,2,2], '3':[1,4,1,1], '4':[1,1,3,2],

      '5':[1,2,3,1], '6':[1,1,1,4], '7':[1,3,1,2], '8':[1,2,1,3], '9':[3,1,1,2]

    };

    const pattern = [...start];

    digits.split('').forEach(d => pattern.push(...map[d], 1));

    pattern.push(...end);


    let isBar = true;

    pattern.forEach((unit, idx) => {

      const span = document.createElement('span');

      span.className = 'barcode-bar';

      span.style.width = `${unit * 2.2}px`;

      span.style.background = isBar ? 'var(--ink)' : 'transparent';

      barcodeEl.appendChild(span);

      isBar = !isBar;

    });

  }


  function syncStoreBindings() {

    const store = sanitizeStoreName(storeNameEl.textContent);

    const upper = store.toUpperCase();

    const slug = slugify(store);

    autoSubline.textContent = `${upper} MART · SHOPPING RECEIPT`;

    autoFooterTitle.textContent = `${upper} MART`;

    autoReceiptCode.textContent = `RCPT-${slug}-001`;

    autoFooterCode.textContent = `NO. ${slug}-0000-0000`;

    const barcodeDigits = makeBarcodeNumber(store);

    barcodeText.textContent = barcodeDigits.replace(/(\d{2})(?=\d)/g, '$1 ');

    buildBarcodeBars(barcodeDigits);

  }


  storeNameEl.addEventListener('input', syncStoreBindings);

  syncStoreBindings();


  document.querySelectorAll('[contenteditable="true"]').forEach(el => {

      if (e.key === 'Enter' && multiline && !e.shiftKey) {

        e.preventDefault();

        document.execCommand('insertLineBreak');

      }

    });

  });


  const palettes = ['#f5f4ef','#d7e7f6','#d8eedf','#f2dce2','#e3ddf6','#f0e4cf','#dddddd','#fbfbf8'];

  function updatePaper(color) {

    document.documentElement.style.setProperty('--paper', color);

  }


  palettes.forEach((color, i) => {

    const btn = document.createElement('button');

    btn.className = 'swatch' + (i === 0 ? ' active' : '');

    btn.type = 'button';

    btn.addEventListener('click', () => {

      if (active) active.classList.remove('active');

      btn.classList.add('active');

      active = btn;

      updatePaper(color);

    });

    if (i === 0) active = btn;

    swatches.appendChild(btn);

  });

    if (active) active.classList.remove('active');

    active = null;

    updatePaper(e.target.value);
